268 --- Set all members of the ip header.
269 --- Per default, all members are set to default values specified in the respective set function.
270 --- Optional named arguments can be used to set a member to a user-provided value.
271 --- @param args Table of named arguments. Available arguments: HardwareAddressType, ProtoAddressType, HardwareAddressLength, ProtoAddressLength, Operation, HardwareSrc, HardwareDst, ProtoSrc, ProtoDst
272 --- @param pre prefix for namedArgs. Default 'arp'.
273 --- @code
274 --- fill() --- only default values
275 --- fill{ arpOperation=2, ipTTL=100 } --- all members are set to default values with the exception of arpOperation
276 --- @endcode
277 function arpHeader:fill(args, pre)
278  args = args or {}
279  pre = pre or "arp"
280 
281  self:setHardwareAddressType(args[pre .. "HardwareAddressType"])
282  self:setProtoAddressType(args[pre .. "ProtoAddressType"])
283  self:setHardwareAddressLength(args[pre .. "HardwareAddressLength"])
284  self:setProtoAddressLength(args[pre .. "ProtoAddressLength"])
285  self:setOperation(args[pre .. "Operation"])
286 
287  local hwSrc = pre .. "HardwareSrc"
288  local hwDst = pre .. "HardwareDst"
289  local prSrc = pre .. "ProtoSrc"
290  local prDst = pre .. "ProtoDst"
291  args[hwSrc] = args[hwSrc] or "01:02:03:04:05:06"
292  args[hwDst] = args[hwDst] or "07:08:09:0a:0b:0c"
293  args[prSrc] = args[prSrc] or "0.1.2.3"
294  args[prDst] = args[prDst] or "4.5.6.7"
295 
296  -- if for some reason the address is in 'struct mac_address'/'union ipv4_address' format, cope with it
297  if type(args[hwSrc]) == "string" then
298  self:setHardwareSrcString(args[hwSrc])
299  else
300  self:setHardwareSrc(args[hwSrc])
301  end
302  if type(args[hwDst]) == "string" then
303  self:setHardwareDstString(args[hwDst])
304  else
305  self:setHardwareDst(args[hwDst])
306  end
307 
308  if type(args[prSrc]) == "string" then
309  self:setProtoSrcString(args[prSrc])
310  else
311  self:setProtoSrc(args[prSrc])
312  end
313  if type(args[prDst]) == "string" then
314  self:setProtoDstString(args[prDst])
315  else
316  self:setProtoDst(args[prDst])
317  end
318 end

345  .. " (hln " .. self:getHardwareAddressLengthString()

407 --- Arp handler task, responds to ARP queries for given IPs and performs arp lookups
408 --- @todo TODO implement garbage collection/refreshing entries \n
409 --- the current implementation does not handle large tables efficiently \n
410 --- @todo TODO multi-NIC support
411 arp.arpTask = "__MG_ARP_TASK"
412 
413 --- Arp table
414 --- TODO docu
415 local arpTable = ns:get()

417 --- Arp handler task
418 --- @param qs TODO
419 --- @todo TODO docu
420 local function arpTask(qs)
421  -- two ways to call this: single nic or array of nics
422  if qs[1] == nil and qs.rxQueue then
423  return arpTask({ qs })
424  end
425 
426  local ipToMac = {}
427  -- loop over NICs/Queues
428  for _, nic in pairs(qs) do
429  if nic.txQueue.dev ~= nic.rxQueue.dev then
430  error("both queues must belong to the same device")
431  end
432 
433  if type(nic.ips) == "string" then
434  nic.ips = { nic.ips }
435  end
436 
437  for _, ip in pairs(nic.ips) do
438  ipToMac[parseIPAddress(ip)] = nic.txQueue.dev:getMac()
439  end
440  nic.txQueue.dev:l2Filter(eth.TYPE_ARP, nic.rxQueue)
441  end
442 
443  local rxBufs = memory.createBufArray(1)
444  local txMem = memory.createMemPool(function(buf)
445  buf:getArpPacket():fill{
446  arpOperation = arp.OP_REPLY,
447  pktLength = 60
448  }
449  end)
450  local txBufs = txMem:bufArray(1)
451 
452  arpTable.taskRunning = true
453 
454  while dpdk.running() do
455 
456  for _, nic in pairs(qs) do
457  rx = nic.rxQueue:tryRecvIdle(rxBufs, 1000)
458  assert(rx <= 1)
459  if rx > 0 then
460  local rxPkt = rxBufs[1]:getArpPacket()
461  if rxPkt.eth:getType() == eth.TYPE_ARP then
462  if rxPkt.arp:getOperation() == arp.OP_REQUEST then
463  local ip = rxPkt.arp:getProtoDst()
464  local mac = ipToMac[ip]
465  if mac then
466  txBufs:alloc(60)
467  -- TODO: a single-packet API would be nice for things like this
468  local pkt = txBufs[1]:getArpPacket()
469  pkt.eth:setSrc(mac)
470  pkt.eth:setDst(rxPkt.eth:getSrc())
471  pkt.arp:setOperation(arp.OP_REPLY)
472  pkt.arp:setHardwareDst(rxPkt.arp:getHardwareSrc())
473  pkt.arp:setHardwareSrc(mac)
474  pkt.arp:setProtoDst(rxPkt.arp:getProtoSrc())
475  pkt.arp:setProtoSrc(ip)
476  nic.txQueue:send(txBufs)
477  end
478  elseif rxPkt.arp:getOperation() == arp.OP_REPLY then
479  -- learn from all arp replies we see (arp cache poisoning doesn't matter here)
480  local mac = rxPkt.arp:getHardwareSrcString()
481  local ip = rxPkt.arp:getProtoSrcString()
482  arpTable[tostring(parseIPAddress(ip))] = { mac = mac, timestamp = dpdk.getTime() }
483  end
484  end
485  rxBufs:freeAll()
486  end
487  end
488 
489  -- send outstanding requests
490  arpTable:forEach(function(ip, value)
491  -- TODO: refresh or GC old entries
492  if value ~= "pending" then
493  return
494  end
495  arpTable[ip] = "requested"
496  -- TODO: the format should be compatible with parseIPAddress
497  ip = tonumber(ip)
498  txBufs:alloc(60)
499  local pkt = txBufs[1]:getArpPacket()
500  pkt.eth:setDstString(eth.BROADCAST)
501  pkt.arp:setOperation(arp.OP_REQUEST)
502  pkt.arp:setHardwareDstString(eth.BROADCAST)
503  pkt.arp:setProtoDst(ip)
504  -- TODO: do not send requests on all devices, but only the relevant
505  for _, nic in pairs(qs) do
506  local mac = nic.txQueue.dev:getMac()
507  pkt.eth:setSrc(mac)
508  pkt.arp:setProtoSrc(parseIPAddress(nic.ips[1]))
509  pkt.arp:setHardwareSrc(mac)
510  nic.txQueue:send(txBufs)
511  end
512  end)
513  dpdk.sleepMillisIdle(1)
514  end
515 end

517 --- Perform a lookup in the ARP table.
518 --- Lookup the MAC address for a given IP.
519 --- Blocks for up to 1 second if the arp task is not yet running
520 --- Caution: this function uses locks and namespaces, must not be used in the fast path
521 --- @param ip The ip address in string or cdata format to look up.
522 function arp.lookup(ip)
523  if type(ip) == "string" then
524  ip = parseIPAddress(ip)
525  elseif type(ip) == "cdata" then
526  ip = ip:get()
527  end
528  if not arpTable.taskRunning then
529  local waitForArpTask = 0
530  while not arpTable.taskRunning and waitForArpTask < 10 do
531  dpdk.sleepMillis(100)
532  end
533  if not arpTable.taskRunning then
534  error("ARP task is not running")
535  end
536  end
537  local mac = arpTable[tostring(ip)]
538  if type(mac) == "table" then
539  return mac.mac, mac.timestamp
540  end
541  arpTable.lock(function()
542  if not arpTable[tostring(ip)] then
543  arpTable[tostring(ip)] = "pending"
544  end
545  end)
546  return nil
547 end
548 
549 --- Perform a non-blocking lookup in the ARP table.
550 --- @param ip The ip address in string or cdata format to look up.
551 --- @param timeout TODO
552 --- @todo FIXME: this only sends a single request
553 function arp.blockingLookup(ip, timeout)
554  local timeout = dpdk.getTime() + timeout
555  repeat
556  local mac, ts = arp.lookup(ip)
557  if mac then
558  return mac, ts
559  end
560  dpdk.sleepMillisIdle(1000)
561  until dpdk.getTime() >= timeout
562 end
