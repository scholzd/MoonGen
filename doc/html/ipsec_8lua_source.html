71 -- Helper function to set the bits (MSB) from..to (LSB)
72 function set_bits32(reg32, from, to, value)

97 -- Enable the Hardware Crypto Engine
98 -- This function must be called before using any other IPSec functions
99 -- @port The port/interface to use
100 function mod.enable(port)
101  --print("IPsec enable, port: "..port)
102  --dump_regs(port)
103 
104  -- Stop TX data path (set TX_DIS bit)
105  local SECTXCTRL__VALUE = set_bit32(dpdkc.read_reg32(port, SECTXCTRL), 1) --set TX_DIS
106  dpdkc.write_reg32(port, SECTXCTRL, SECTXCTRL__VALUE)
107 
108  -- Stop RX data path (set RX_DIS bit)
109  local SECRXCTRL__VALUE = set_bit32(dpdkc.read_reg32(port, SECRXCTRL), 1) --set RX_DIS
110  dpdkc.write_reg32(port, SECRXCTRL, SECRXCTRL__VALUE)
111 
112  -- Wait for the data paths to be emptied by hardware (check SECTX/RX_RDY bits).
113  repeat
114  --print("Waiting for registers to be asserted by hardware...")
115  dpdk.sleepMillis(100) -- wait 100ms before poll
116  local SECTXSTAT__SECTX_RDY = bit.band(dpdkc.read_reg32(port, SECTXSTAT), 0x1)
117  local SECRXSTAT__SECRX_RDY = bit.band(dpdkc.read_reg32(port, SECRXSTAT), 0x1)
118  --print("SECTX_RDY: "..SECTXSTAT__SECTX_RDY..", SECRX_RDY: "..SECRXSTAT__SECRX_RDY)
119  until SECTXSTAT__SECTX_RDY == 0x1 and SECRXSTAT__SECRX_RDY == 0x1
120 
121  -- Set MINSECIFG to 0x3
122  local SECTXMINIFG__VALUE = set_bits32(dpdkc.read_reg32(port, SECTXMINIFG), 3, 0, 0x3) --set MINSECIFG
123  dpdkc.write_reg32(port, SECTXMINIFG, SECTXMINIFG__VALUE)
124 
125  -- Set FULLTHRESH to 0x15
126  local SECTXBUFFAF__VALUE = set_bits32(dpdkc.read_reg32(port, SECTXBUFFAF), 9, 0, 0x15) --set FULLTHRESH
127  dpdkc.write_reg32(port, SECTXBUFFAF, SECTXBUFFAF__VALUE)
128 
129  -- Enable TX crypto engine
130  local SECTXCTRL__VALUE = clear_bit32(dpdkc.read_reg32(port, SECTXCTRL), 0) --clear SECTX_DIS
131  SECTXCTRL__VALUE = set_bit32(SECTXCTRL__VALUE, 2) --set STORE_FORWARD
132  dpdkc.write_reg32(port, SECTXCTRL, SECTXCTRL__VALUE)
133 
134  -- Enable RX crypto engine
135  local SECRXCTRL__VALUE = clear_bit32(dpdkc.read_reg32(port, SECRXCTRL), 0) --clear SECRX_DIS
136  dpdkc.write_reg32(port, SECRXCTRL, SECRXCTRL__VALUE)

144  dpdkc.write_reg32(port, IPSRXIDX, IPSRXIDX__VALUE)

148  dpdkc.write_reg32(port, SECTXCTRL, SECTXCTRL__VALUE)

215 -- Write AES 128 bit Key and Salt into the Hardware TX SA table
216 -- @port: The port/interface to use
217 -- @idx: Index into TX SA table (0-1023)
218 -- @key: 128 bit AES key (as hex string)
219 -- @salt: 32 bit AES salt (as hex string)
220 function mod.tx_set_key(port, idx, key, salt)
221  if idx > 1023 or idx < 0 then
222  error("Idx must be in range 0-1023")
223  end
224  if string.len(key) ~= 32 then
225  error("Key must be 128 bit (hex string).")
226  end
227  if string.len(salt) ~= 8 then
228  error("Salt must be 32 bit (hex string).")
229  end
230 
231  local key_3 = tonumber(string.sub(key, 1, 8), 16) --MSB
232  local key_2 = tonumber(string.sub(key, 9, 16), 16)
233  local key_1 = tonumber(string.sub(key, 17, 24), 16)
234  local key_0 = tonumber(string.sub(key, 25, 32), 16) --LSB
235  local _salt = tonumber(salt, 16)
236 
237  -- Prepare command to write key to SA_IDX
238  local value = set_bits32(dpdkc.read_reg32(port, IPSTXIDX), 12, 3, idx) --set SA_IDX
239  value = clear_bit32(value, 30) --clear READ
240  value = set_bit32(value, 31) --set WRITE
241  --print("IPSTXIDX: 0x"..uhex32(value))
242 
243  --prepare registers
244  dpdkc.write_reg32(port, IPSTXKEY_3, key_3)
245  dpdkc.write_reg32(port, IPSTXKEY_2, key_2)
246  dpdkc.write_reg32(port, IPSTXKEY_1, key_1)
247  dpdkc.write_reg32(port, IPSTXKEY_0, key_0)
248  dpdkc.write_reg32(port, IPSTXSALT, _salt)
249  --push to hw
250  dpdkc.write_reg32(port, IPSTXIDX, value)
251  --pass SA_IDX via 'TX context descriptor' to use this SA!
252 end

284 -- Write AES 128 bit Key and Salt into the Hardware RX SA table
285 -- @port: the port/interface to use
286 -- @idx: Index into SA RX table (0-1023)
287 -- @key: 128 bit AES key (as hex string)
288 -- @salt: 32 bit AES salt (as hex string)
289 -- @ip_ver: IP Version for which this SA is valid (4 or 6)
290 -- @proto: IPSec protocol type to use ("esp" or "ah")
291 -- @decrypt: ESP mode (1=ESP decrypt and authenticate, 0=ESP authenticate only)
292 function mod.rx_set_key(port, idx, key, salt, ip_ver, proto, decrypt)
293  if idx > 1023 or idx < 0 then
294  error("Idx must be in range 0-1023")
295  end
296  if string.len(key) ~= 32 then
297  error("Key must be 128 bit (hex string).")
298  end
299  if string.len(salt) ~= 8 then
300  error("Salt must be 32 bit (hex string).")
301  end
302 
303  local ipv6 = nil
304  if ip_ver == 4 then
305  ipv6 = 0
306  elseif ip_ver == 6 then
307  ipv6 = 1
308  else
309  error("IP version must be either 4 or 6")
310  end
311 
312  local esp = nil
313  if proto == "esp" then
314  esp = 1
315  elseif proto == "ah" then
316  esp = 0
317  else
318  error("Protocol must be either 'esp' or 'ah'")
319  end
320 
321  local esp_mode = decrypt or 0
322  if esp_mode ~= 1 and esp_mode ~= 0 then
323  error("ESP Decrypt must be either 0 or 1")
324  end
325 
326  local key_3 = tonumber(string.sub(key, 1, 8), 16) --MSB
327  local key_2 = tonumber(string.sub(key, 9, 16), 16)
328  local key_1 = tonumber(string.sub(key, 17, 24), 16)
329  local key_0 = tonumber(string.sub(key, 25, 32), 16) --LSB
330  local _salt = tonumber(salt, 16)
331 
332  -- Prepare command to write KEY_TABLE at TB_IDX
333  local value = set_bits32(dpdkc.read_reg32(port, IPSRXIDX), 2, 1, 0x3) --set TABLE to KEY(0x3)
334  value = set_bits32(value, 12, 3, idx) --set TB_IDX
335  value = clear_bit32(value, 30) --clear READ
336  value = set_bit32(value, 31) --set WRITE
337  --print("IPSRXIDX: 0x"..uhex32(value))
338 
339  -- Prepare security mode register
340  local mode = set_bit32(dpdkc.read_reg32(port, IPSRXMOD), 0) --set VALID, 1=valid, 0=invalid SA
341  mode = set_bits32(mode, 2, 2, esp) --set PROTO, 1=ESP, 0=AH
342  mode = set_bits32(mode, 3, 3, esp_mode) --set DECRYPT, 1=decrypt(ESP), 0=authenticate(ESP)
343  mode = set_bits32(mode, 4, 4, ipv6) --set IPV6, 1=IPv6, 0=IPv4
344 
345  --prepare registers
346  dpdkc.write_reg32(port, IPSRXKEY_3, key_3)
347  dpdkc.write_reg32(port, IPSRXKEY_2, key_2)
348  dpdkc.write_reg32(port, IPSRXKEY_1, key_1)
349  dpdkc.write_reg32(port, IPSRXKEY_0, key_0)
350  dpdkc.write_reg32(port, IPSRXSALT, _salt)
351  dpdkc.write_reg32(port, IPSRXMOD, mode)
352  --push to hw
353  dpdkc.write_reg32(port, IPSRXIDX, value)
354 end

356 -- Read AES 128 bit Key and Salt from the Hardware RX SA table
357 -- @port: The port/interface to use
358 -- @idx: Index into the RX SA table (0-1023)
359 -- @return: Key and Salt (as hex string),
360 -- Valid Flag (1=SA is valid, 0=SA is invalid)
361 -- Proto Flag (1=ESP, 0=AH)
362 -- Decrypt Flag (1=ESP decrypt and authenticate, 0=ESP authenticate only)
363 -- IPv6 Flag (1=SA is valid for IPv6, 0=SA is valid for IPv4)
364 function mod.rx_get_key(port, idx)
365  if idx > 1023 or idx < 0 then
366  error("Idx must be in range 0-1023")
367  end
368 
369  -- Prepare command to read from KEY_TABLE at TB_IDX
370  local value = set_bits32(dpdkc.read_reg32(port, IPSRXIDX), 2, 1, 0x3) --set TABLE to KEY(0x3)
371  value = set_bits32(value, 12, 3, idx) --set TB_IDX
372  value = set_bit32(value, 30) --set READ
373  value = clear_bit32(value, 31), --clear WRITE
374  --print("IPSRXIDX: 0x"..uhex32(value))
375 
376  --pull from hw
377  dpdkc.write_reg32(port, IPSRXIDX, value)
378 
379  --fetch result
380  local key_3 = dpdkc.read_reg32(port, IPSRXKEY_3)
381  local key_2 = dpdkc.read_reg32(port, IPSRXKEY_2)
382  local key_1 = dpdkc.read_reg32(port, IPSRXKEY_1)
383  local key_0 = dpdkc.read_reg32(port, IPSRXKEY_0)
384  local _salt = dpdkc.read_reg32(port, IPSRXSALT)
385  local _mode = dpdkc.read_reg32(port, IPSRXMOD)
386 
387  local valid = bit.rshift(bit.band(_mode, 0x01), 0)
388  local proto = bit.rshift(bit.band(_mode, 0x04), 2)
389  local decrypt = bit.rshift(bit.band(_mode, 0x08), 3)
390  local ipv6 = bit.rshift(bit.band(_mode, 0x10), 4)
391  local key = uhex32(key_3)..uhex32(key_2)..uhex32(key_1)..uhex32(key_0)
392 
393  return key, uhex32(_salt), valid, proto, decrypt, ipv6
394 end

437 -- Read IP-Address from the Hardware RX IP table
438 -- @port: The port/interface to use
439 -- @idx: Index into the RX IP table (0-127)
440 -- @is_ipv4: IP Version expected (true/false)
441 -- @return: The IP(v4/v6)-Address (as string) and a IP Version Flag (true=IPv4, false=IPv6)
442 function mod.rx_get_ip(port, idx, is_ipv4)
443  if idx > 127 or idx < 0 then
444  error("Idx must be in range 0-127")
445  end
446  if is_ipv4 == nil then
447  is_ipv4 = true
448  end
449 
450  -- Prepare command to read from IP_TABLE at TB_IDX
451  local value = set_bits32(dpdkc.read_reg32(port, IPSRXIDX), 2, 1, 0x1) --set TABLE to IP(0x1)
452  value = set_bits32(value, 12, 3, idx) --set TB_IDX
453  value = set_bit32(value, 30) --set READ
454  value = clear_bit32(value, 31) --clear WRITE
455  --print("IPSRXIDX: 0x"..uhex32(value))
456 
457  --pull from hw
458  dpdkc.write_reg32(port, IPSRXIDX, value)
459  --fetch result
460  local ip_3 = dpdkc.read_reg32(port, IPSRXIPADDR_3)
461  local ip_2 = dpdkc.read_reg32(port, IPSRXIPADDR_2)
462  local ip_1 = dpdkc.read_reg32(port, IPSRXIPADDR_1)
463  local ip_0 = dpdkc.read_reg32(port, IPSRXIPADDR_0)
464 
465  local ip = nil
466  if is_ipv4 == true then
467  local ip4 = ffi.new("union ip4_address")
468  ip4.uint32 = ip_3
469  ip = ip4:getString()
470  else
471  local ip6 = ffi.new("union ip6_address")
472  ip6.uint32[3] = ip_3
473  ip6.uint32[2] = ip_2
474  ip6.uint32[1] = ip_1
475  ip6.uint32[0] = ip_0
476  ip = ip6:getString()
477  end
478 
479  return ip, is_ipv4

511  dpdkc.write_reg32(port, IPSRXIPIDX, ip_idx)
